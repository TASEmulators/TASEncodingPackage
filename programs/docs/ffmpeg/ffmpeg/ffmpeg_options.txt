# Required stuff
--disable-autodetect
--enable-bzlib
--enable-lzma
--enable-zlib

# Disable everything and manually enable only what we want
--disable-avdevice
--disable-swscale
--disable-postproc
--disable-network
--disable-schannel
--disable-everything
--disable-debug

# Basics and Avisynth
--enable-protocol=file
--enable-protocol=pipe
--enable-avisynth
--enable-demuxer=avisynth
--enable-decoder=rawvideo
--enable-decoder=pcm*

# Null output stuff for generating timecodes.
--enable-muxer=null
--enable-encoder=wrapped_avframe

# Audio encoding
--enable-libfdk-aac
--enable-encoder=libfdk_aac
--enable-encoder=flac
--enable-muxer=mp4
--enable-muxer=flac
--enable-filter=aresample       # For resampling between formats
--enable-filter=atrim           # For fixing AAC priming offset
